# Symbol

SymbolжҳҜ ES6 ж–°еј•е…Ҙзҡ„еҺҹе§Ӣж•°жҚ®зұ»еһӢ

Symbolе®һдҫӢжҳҜе”ҜдёҖпјҢдёҚеҸҜеҸҳзҡ„

# еҹәжң¬з”Ёжі•

let s = Symbol()

console.log(typeof s) // symbol

жіЁж„Ҹ

Symbol()еҮҪж•°еүҚдёҚиғҪдҪҝз”Ёnewе‘Ҫд»ӨпјҢеҗҰеҲҷдјҡжҠҘй”ҷ

SymbolжҳҜдёҖдёӘеҺҹе§Ӣзұ»еһӢзҡ„еҖј,дёҚжҳҜеҜ№иұЎ,жүҖд»ҘдёҚиғҪдҪҝз”Ёnewе‘Ҫд»ӨжқҘи°ғз”Ё

SymbolеҖјдёҚжҳҜеҜ№иұЎпјҢжүҖд»Ҙд№ҹдёҚиғҪж·»еҠ еұһжҖ§

и°ғз”ЁSymbol()еҮҪж•°ж—¶еҸҜд»Ҙдј е…ҘдёҖдёӘеӯ—з¬ҰдёІеҸӮж•°дҪңдёәжҸҸиҝ°(description)

дё»иҰҒжҳҜдёәдәҶеңЁжҺ§еҲ¶еҸ°жҳҫзӨәпјҢжҲ–иҖ…иҪ¬дёәеӯ—з¬ҰдёІж—¶пјҢжҜ”иҫғе®№жҳ“еҢәеҲҶгҖӮ

let genericSymbol = Symbol()
let otherGenericSymbol = Symbol()

let fooSymbol = Symbol('foo')
let otherFooSymbol = Symbol('foo')

console.log(genericSymbol === otherFooSymbol) // false
console.log(fooSymbol === otherFooSymbol) // false

иҜҙжҳҺ

Symbol()еҸӮж•°еҸӘдҪңдёәжҸҸиҝ°

зӣёеҗҢеҸӮж•°зҡ„иҝ”еӣһеҖјдёҚзӣёзӯү

еҰӮжһңSymbolзҡ„еҸӮж•°жҳҜдёҖдёӘеҜ№иұЎпјҢе°ұдјҡи°ғз”ЁиҜҘеҜ№иұЎзҡ„toString()ж–№жі•пјҢе°Ҷе…¶иҪ¬дёәеӯ—з¬ҰдёІпјҢз„¶еҗҺжүҚз”ҹжҲҗдёҖдёӘSymbolеҖј

const obj = {
  toString() {
    return 'hello'
  }
}
let sym = Symbol(obj)
console.log(sym) // Symbol(hello)

ES2019 жҸҗдҫӣдәҶе®һдҫӢеұһжҖ§descriptionзӣҙжҺҘиҝ”еӣһSymbolеҖјзҡ„жҸҸиҝ°

let fooSymbol = Symbol('foo')

console.log(fooSymbol.description) // foo

# дҪңдёәеұһжҖ§еҗҚзҡ„Symbol

Symbolз”ЁдәҺеҜ№иұЎзҡ„еұһжҖ§еҗҚпјҢе°ұиғҪдҝқиҜҒдёҚдјҡеҮәзҺ°еҗҢеҗҚзҡ„еұһжҖ§

let mySymbol = Symbol()

let a = {}
a[mySymbol] = 'hello!'

let b = {
  [mySymbol]: 'hello!'
}

let c = {}
Object.defineProperty(c, mySymbol, { value: 'hello!' })

console.log(a[mySymbol]) // hello!
console.log(b[mySymbol]) // hello!
console.log(c[mySymbol]) // hello!

жіЁж„Ҹ

SymbolдҪңдёәеҜ№иұЎеұһжҖ§еҗҚдёҚиғҪз”ЁзӮ№иҝҗз®—з¬Ұ

дҪҝз”ЁSymbolеҖје®ҡд№үеұһжҖ§ж—¶пјҢSymbolеҖјеҝ…йЎ»ж”ҫеңЁж–№жӢ¬еҸ·д№Ӣдёӯ

# е®һдҫӢпјҡж¶ҲйҷӨйӯ”жңҜеӯ—з¬ҰдёІ

йӯ”жңҜеӯ—з¬ҰдёІпјҡеңЁд»Јз Ғд№ӢдёӯеӨҡж¬ЎеҮәзҺ°гҖҒдёҺд»Јз ҒеҪўжҲҗејәиҖҰеҗҲзҡ„жҹҗдёҖдёӘе…·дҪ“зҡ„еӯ—з¬ҰдёІжҲ–иҖ…ж•°еҖј

function getArea(shape, options) {
  let area = 0
  switch (shape) {
    case 'Triangle': // йӯ”жңҜеӯ—з¬ҰдёІ
      area = 0.5 * options.width * options.height
      break

    default:
      break
  }
  return area
}
getArea('Triangle', { width: 100, height: 100 }) // йӯ”жңҜеӯ—з¬ҰдёІ

ж¶ҲйҷӨж–№жі•пјҡе°ҶTriangleи®ҫдёәеҸҳйҮҸ

const shapeType = {
  triangle: Symbol('triangle')
}
function getArea(shape, options) {
  let area = 0
  switch (shape) {
    case shapeType.triangle:
      area = 0.5 * options.width * options.height
      break

    default:
      break
  }
  return area
}
getArea(shapeType.triangle, { width: 100, height: 100 })

# еұһжҖ§еҗҚйҒҚеҺҶ

SymbolдҪңдёәеҜ№иұЎеұһжҖ§еұһжҖ§ж—¶з”ЁObject.getOwnPropertySymbols()ж–№жі•пјҢиҺ·еҸ–еҜ№иұЎдёӯжүҖжңүSymbolеұһжҖ§еҗҚпјҢиҝ”еӣһдёҖдёӘж•°з»„

Object.getOwnPropertyNames()иҝ”еӣһеёёи§„еұһжҖ§зҡ„ж•°з»„

Object.getOwnPropertyDescriptors(o)иҝ”еӣһеҗҢж—¶еҢ…еҗ«еёёи§„е’Ң Symbol еұһжҖ§зҡ„еҜ№иұЎ

Reflect.ownKeys() ж–№жі•еҸҜд»Ҙиҝ”еӣһжүҖжңүзұ»еһӢзҡ„й”®еҗҚпјҢеҢ…жӢ¬еёёи§„й”®еҗҚе’Ң Symbol й”®еҗҚзҡ„ж•°з»„

let s1 = Symbol('foo')
let s2 = Symbol('bar')

let o = {
  [s1]: 'foo val',
  [s2]: 'bar val',
  baz: 'baz val',
  qux: 'qux val'
}

console.log(Object.getOwnPropertySymbols(o))
// [Symbol(foo), Symbol(bar)]

console.log(Object.getOwnPropertyNames(o))
// ['baz', 'qux']

console.log(Object.getOwnPropertyDescriptors(o))
// {baz: {вҖҰ}, qux: {вҖҰ}, Symbol(foo): {вҖҰ}, Symbol(bar): {вҖҰ}}

console.log(Reflect.ownKeys(o))
//  ['baz', 'qux', Symbol(foo), Symbol(bar)]

# Symbol.for()е’ҢSymbol.keyFor()

Symbol.for()жҺҘеҸ—дёҖдёӘеӯ—з¬ҰдёІеҸӮж•°пјҢдјҡе…ҲжҗңзҙўжңүжІЎжңүд»ҘиҜҘеӯ—з¬ҰдёІдҪңдёәеҗҚз§°зҡ„Symbol,жңүе°ұиҝ”еӣһиҝҷдёӘеҖјпјҢжІЎжңүе°ұж–°е»әдёҖдёӘе№¶жіЁеҶҢеҲ°е…ЁеұҖ

let s1 = Symbol.for('foo') // еҲӣе»әж–°з¬ҰеҸ·
let s2 = Symbol.for('foo') // йҮҚз”Ёе·Іжңүзҡ„з¬ҰеҸ·

console.log(s1 === s2) // true

Symbol.keyFor()иҝ”еӣһдёҖдёӘе·Ізҷ»и®°зҡ„ Symbol зұ»еһӢеҖјзҡ„key

let s1 = Symbol('foo')
let s2 = Symbol.for('foo')

console.log(Symbol.keyFor(s1)) // undefined
console.log(Symbol.keyFor(s2)) // foo
